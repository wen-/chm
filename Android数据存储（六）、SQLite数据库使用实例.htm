三、实现MVC，封装业务方法，成为Controller

[java] view plaincopy
  1. package com.tao.sqlitedb;  
  2.   
  3. import java.util.ArrayList;  
  4. import java.util.List;  
  5.   
  6. import android.content.Context;  
  7. import android.database.Cursor;  
  8. import android.database.sqlite.SQLiteDatabase;  
  9.   
  10. public class DBManager {  
  11.     private DBOpenHelper helper;  
  12.     private SQLiteDatabase db;  
  13.   
  14.     public DBManager(Context context) {  
  15.         helper = new DBOpenHelper(context);  
  16.         // 因为getWritableDatabase内部调用了mContext.openOrCreateDatabase(mName, 0,  
  17.         // mFactory);  
  18.         // 所以要确保context已初始化,我们可以把实例化DBManager的步骤放在Activity的onCreate里  
  19.         db = helper.getWritableDatabase();  
  20.     }  
  21.   
  22.     /** 
  23.      *  
  24.      * 添加一条记录 
  25.      *  
  26.      * @param person 
  27.      */  
  28.     public void add(Person person) {  
  29.         db.execSQL("insert into tbl_person values(null,?,?)", new Object[] { person.getName(), person.getAge() });  
  30.   
  31.     }  
  32.   
  33.     /** 
  34.      * 添加多条记录 
  35.      *  
  36.      * @param persons 
  37.      */  
  38.     public void adds(List<Person> persons) {  
  39.         db.beginTransaction();  
  40.         try {  
  41.             for (Person p : persons) {  
  42.                 db.execSQL("insert into tbl_person values(null,?,?)", new Object[] { p.getName(), p.getAge() });  
  43.             }  
  44.             db.setTransactionSuccessful();  
  45.         } catch (Exception e) {  
  46.             e.printStackTrace();  
  47.         } finally {  
  48.             db.endTransaction();  
  49.         }  
  50.     }  
  51.   
  52.     /** 
  53.      * 更新一条记录 
  54.      *  
  55.      * @param person 
  56.      */  
  57.     public void update(Person person) {  
  58.         db.execSQL("update tbl_person set name=?,age=? where _id=?", new Object[] { person.getName(), person.getAge(), person.get_id() });  
  59.     }  
  60.   
  61.     /** 
  62.      * 删除一条记录 
  63.      *  
  64.      * @param person 
  65.      */  
  66.     public void delete(int id) {  
  67.         db.execSQL("delete from tbl_person where _id=?", new Object[] { id });  
  68.     }  
  69.   
  70.     /** 
  71.      * 查询一条记录 
  72.      */  
  73.     public Person queryOne(int _id) {  
  74.         Person person = new Person();  
  75.         Cursor c = db.rawQuery("select * from tbl_person where _id=?", new String[] { _id + "" });  
  76.         while (c.moveToNext()) {  
  77.             person.set_id(c.getInt(c.getColumnIndex("_id")));  
  78.             person.setName(c.getString(c.getColumnIndex("name")));  
  79.             person.setAge(c.getInt(c.getColumnIndex("age")));  
  80.         }  
  81.         c.close();  
  82.         return person;  
  83.     }  
  84.   
  85.     /** 
  86.      * 查询多条记录 
  87.      *  
  88.      * @return List<Person> 
  89.      */  
  90.     public List<Person> queryMany() {  
  91.         ArrayList<Person> persons = new ArrayList<Person>();  
  92.         Cursor c = db.rawQuery("select * from tbl_person", null);  
  93.         while (c.moveToNext()) {  
  94.             Person person = new Person();  
  95.             person.set_id(c.getInt(c.getColumnIndex("_id")));  
  96.             person.setName(c.getString(c.getColumnIndex("name")));  
  97.             person.setAge(c.getInt(c.getColumnIndex("age")));  
  98.             persons.add(person);  
  99.         }  
  100.         c.close();  
  101.         return persons;  
  102.     }  
  103.   
  104. }

  20. public class SqliteDBActivity extends Activity {  
  21.     private EditText ed1;  
  22.     private EditText ed2;  
  23.     private EditText ed3;  
  24.     private ListView listView;

  26.     @Override  
  27.     public void onCreate(Bundle savedInstanceState) {

  29.         setContentView(R.layout.main);

  36.     public void add(View view){

  39.     }  
  40.       
  41.     public void delete(View view){

  44.       
  45.     public void update(View view){  
  46.         Person person=new Person(Integer.parseInt(ed3.getText().toString().trim()),ed1.getText().toString().trim(), Integer.parseInt(ed2.getText().toString().trim()));

  48.     }  
  49.       
  50.     public void queryOne(View view){

  52.         List<Person> persons=new ArrayList<Person>();  
  53.         persons.add(person);  
  54.         setListView(persons);

  56.       
  57.     public void queryMany(View view){  
  58.         List<Person> persons=dbManager.queryMany();

  60.     }  
  61.       
  62.     public void setListView(List<Person> persons){  
  63.         List<Map<String, Object>> list=new ArrayList<Map<String,Object>>();  
  64.         for(Person p:persons){  
  65.             Map<String, Object> map=new HashMap<String, Object>();  
  66.             map.put("id", p.get_id());  
  67.             map.put("name", p.getName());  
  68.             map.put("age", p.getAge());  
  69.             list.add(map);  
  70.         }  
  71.         SimpleAdapter adapter=new SimpleAdapter(this, list, R.layout.cell, new String[]{"id","name","age"}, new int[]{R.id.text1,R.id.text2,R.id.text3});  
  72.         listView.setAdapter(adapter);  
  73.     }  
  74. }
