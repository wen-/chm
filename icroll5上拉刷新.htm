实现效果：类似网易新闻加载新闻列表（好吧，我的只能算是基础版，如要添加动图或者css样式或者canvas效果请自行脑补）

js需要说明的是，这里的scroll事件当然不是原生dom的scroll事件，而是IScroll对象的滚动事件，类似touchmove事件，scrollEnd事件也类似touchend事件，在滚动结束后执行

[javascript] view plain copy
  1. <script type="text/javascript">  
  2.   
  3.     function load () {  
  4.         var myScroll,  
  5.                     pullDown = $("#pullDown"),  
  6.                     pullUp = $("#pullUp"),  
  7.                     pullDownLabel = $(".pullDownLabel"),  
  8.                     pullUpLabel = $(".pullUpLabel"),  
  9.                     container = $('#list'),  
  10.                     loadingStep = 0;//加载状态0默认，1显示加载状态，2执行加载数据，只有当为0时才能再次加载，这是防止过快拉动刷新  
  11.   
  12.             pullDown.hide();  
  13.             pullUp.hide();  
  14.   
  15.             myScroll = new IScroll("#wrapper", {  
  16.                 scrollbars: true,  
  17.                 mouseWheel: false,  
  18.                 interactiveScrollbars: true,  
  19.                 shrinkScrollbars: 'scale',  
  20.                 fadeScrollbars: true,  
  21.                 scrollY:true,  
  22.                 probeType: 2,  
  23.                 bindToWrapper:true  
  24.             });  
  25.             myScroll.on("scroll",function(){  
  26.                 if(loadingStep == 0 && !pullDown.attr("class").match('refresh|loading') && !pullUp.attr("class").match('refresh')){  
  27.                     if(this.y > 40){//下拉刷新操作  
  28.                         $(".pulldown-tips").hide();  
  29.                         pullDown.addClass("refresh").show();  
  30.                         pullDownLabel.text("松手刷新数据");  
  31.                         loadingStep = 1;  
  32.                         myScroll.refresh();  
  33.                     }else if(this.y < (this.maxScrollY - 14)){//上拉加载更多  
  34.                         pullUp.addClass("refresh").show();  
  35.                         pullUpLabel.text("正在载入");  
  36.                         loadingStep = 1;  
  37.                         pullUpAction();  
  38.                     }  
  39.                 }  
  40.             });  
  41.             myScroll.on("scrollEnd",function(){  
  42.                 if(loadingStep == 1){  
  43.                     if( pullDown.attr("class").match("refresh") ){//下拉刷新操作  
  44.                         pullDown.removeClass("refresh").addClass("loading");  
  45.                         pullDownLabel.text("正在刷新");  
  46.                         loadingStep = 2;  
  47.                         pullDownAction();  
  48.                     }  
  49.                 }  
  50.             });  
  51.   
  52.         function pullDownAction(){  
  53.             setTimeout(function(){  
  54.                 var li, i;  
  55.                 for (i = 0,li = ""; i < 3; i++) {  
  56.                     li += "<li>" + "new Add " + new Date().toLocaleString() + " ！" + "</li>";  
  57.                 }  
  58.                 container.prepend(li);  
  59.                 pullDown.attr('class','').hide();  
  60.                 myScroll.refresh();  
  61.                 loadingStep = 0;  
  62.                 $(".pulldown-tips").show();  
  63.             },1000);  
  64.         }  
  65.         function pullUpAction(){  
  66.             setTimeout(function(){  
  67.                 var li, i;  
  68.                 for (i = 0,li = ""; i < 3; i++) {  
  69.                     li += "<li>" + "new Add " + new Date().toLocaleString() + " ！" + "</li>";  
  70.                 }  
  71.                 container.append(li);  
  72.                 pullUp.attr('class','').hide();  
  73.                 myScroll.refresh();  
  74.                 loadingStep = 0;  
  75.             },1000);  
  76.         }  
  77.   
  78.         document.addEventListener('touchmove', function (e) { e.preventDefault(); }, false);  
  79.         }  
  80.   
  81.   
  82. </script>
