js编码转换（加解密）

escape("abc") //加密

unescape(escape("abc") ) //解密

encodeURI("中国"); //转码只限中文

decodeURI("%E4%B8%AD%E5%9B%BD");//解码

btoa("abc") //Base64转码

atob("YWJj")//Base64解码

 

中文转unicode（其实就是escape后把%u转为\u）

function unicode (str){
    return str.replace(/[^\u0000-\u00FF]/g,function($0){return escape($0).replace(/(%u)(\w{4})/gi," \\u$2")});
};

unicode("中国");

unicode转中文（其实就是把\u转为%u后再unescape）

function cncode(str){
    return unescape(str.replace(/\\u/g, "%u"));
}
