window.onload = function(){
  //根据屏幕尺寸修改项目名称的显示效果
  $(window).resize(function(){
   $("div.sort_live > div").each(function(){
    var $item = $(this).find("span");
    var item_name = $item.text();
    //清除旧的显示效果
    $item.css("width","auto").find("marquee").replaceWith(item_name);
    //设置新的显示效果
    var max_width = $(this).find("a").innerWidth();
    var img_width = $(this).find("img").outerWidth(true);
    var span_width = $item.outerWidth(true);
    var able_width = parseInt((max_width-img_width)*0.96);
    var cut_width = able_width - span_width;
    if (cut_width>0) return true;
    $item.replaceWith("<span><marquee scrollamount=\"2\" behavior=\"alternate\">"+item_name+"</marquee></span>");
    $(this).find("span").css("width", able_width+"px");
   });
  }).resize();
 };
