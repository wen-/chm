<div class="goTop" style="margin-left:525px;">goTop</div>
<style>
.goTop{
 display:none;
 height:50px;
 width:50px;
 background-color:red;
 position:fixed;
 _position:absolute;
 z-index:100;
 bottom:10px;
 left:50%;
 margin-left:525px;
 _top:expression(eval(documentElement.scrollTop + document.documentElement.offsetHeight-80));
}

</style>
<script>
$(window).scroll(function(){
 if($(window).scrollTop() > 200){
  $(".goTop").fadeIn("slow");
 }else if($(window).scrollTop() < 200){
  $(".goTop").fadeOut("slow");
 };
});
$(".goTop").click(function(){
   var sTOP = $(window).scrollTop();
 
    var goTOP = window.setInterval(function(){
  if(sTOP > 0){
   $(window).scrollTop(sTOP);
   sTOP -= 10;
  }else{
   $(window).scrollTop(0);
   window.clearInterval(goTOP);
  };
    },10);
});
</script>
