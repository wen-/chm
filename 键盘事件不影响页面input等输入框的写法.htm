键盘事件不影响页面input等输入框的写法

$(window).bind('keydown', function(e){

  if (e.target && e.target.tagName.toLowerCase()!='input')

//判断键盘事件不是在input标签时才执行下面的代码；


   if (e.keyCode==37)
    $('#book').turn('previous');
   else if (e.keyCode==39)
    $('#book').turn('next');

 });

document.attachEvent("onclick",function(e){//监听event标签
    if (e.srcElement.nodeName.toLowerCase()!='a'){
     var li = div_nav.getElementsByTagName("li");//原先获取节点的方法不支持；
     for(var k=0; k<li.length; k++){   //对子节点进行操作，不管子节点有单个还是多个，都必须遍历操作
      li[k].className = "";
     };
    }

 });
