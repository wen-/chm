通过原型继承一个新对象

function inherit(p){

if(p == null)throw TypeError();

if(Object.create) return Object.create(p);

var t = typeof p;

if(t !== "object" && t !== "function"){

function f(){};
f.prototype = p;
return new f();

}

}

 

属性继承：

var unitcircle = {r:1};

var c = inherit(unitcircle);

c.x =1;c.y = 1;

c.r = 2;        //覆盖继承来的r属性

unitcircle.r;   //=1 原型链不会被修改
