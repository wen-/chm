function showHtml(obj){
        var m_arguments = ["b_bid","b_bid_num","ktje","b_interest_rate_year","b_dateline","b_totle_money","rate","atext"];
        var l = m_arguments.length;
        var txt = $("#moduleHtmlTz").html();
        var htmls = [];
        $.each(obj,function(i,n){
            var ii = 0;
            var htm = txt;
            n.b_interest_rate_year = n.b_interest_rate_year*100;
            for(;ii<l;ii++){
                var reg = new RegExp('\\{\\%'+m_arguments[ii]+'\\%\\}', 'g');
                if(m_arguments[ii] == 'b_bid_num'){
                    htm = htm.replace(reg, addZore(n[m_arguments[ii-1]],5));
                }else if(m_arguments[ii] == 'atext'){
                    if(n[m_arguments[ii]] == 1){
                        htm = htm.replace(reg, '<button type="button" class="btn bg_green boxshadow_green">立即投资</button>');
                    }else if(n[m_arguments[ii]] == 2){
                        htm = htm.replace(reg, '<button type="button" class="btn bg_orange boxshadow_orange">已售完</button>');
                    }else if(n[m_arguments[ii]] == 3){
                        htm = htm.replace(reg, '<button type="button" class="btn bg_brown boxshadow_brown">已结束</button>');
                    }else if(n[m_arguments[ii]] == 4){
                        htm = htm.replace(reg, '<button type="button" class="btn bg_green boxshadow_green">我要预定</button>');
                    }
                }else{
                    htm = htm.replace(reg, n[m_arguments[ii]]);
                }

            }
            htmls.push(htm);
        });
        return htmls;
    }
