   function 分类(){
      var reg = document.getElementById("reg").value;
      var reg1 = document.getElementById("reg1").value;
      var reg2 = document.getElementById("reg2").value;
      var reg3 = document.getElementById("reg3").value;
      var str = document.getElementById("inputNum").value;
      var regexp = new RegExp(reg, "g");
      var reg1exp = new RegExp(reg1);
      var reg2exp = new RegExp(reg2);
      var reg3exp = new RegExp(reg3);
      var numArray = [];
      var num1Array = [];
      var num2Array = [];
      var num3Array = [];
      if(!str){
         alert("请输入要分类的号码");
         return false;
      }
      numArray = str.match(regexp) && str.match(regexp).sort();
      if(!(numArray&&numArray.length)){
         alert("无号码可分类");
         return false;
      }
      //console.log(numArray);
      for(var i=0;i<numArray.length;i++){
         if(numArray[i] == numArray[i+1]){
            continue;
         }
         //console.log(numArray[i]);
         if(reg1exp.test(numArray[i])){
            num1Array.push(numArray[i]);
         }
         if(reg2exp.test(numArray[i])){
            num2Array.push(numArray[i]);
         }
         if(reg3exp.test(numArray[i])){
            num3Array.push(numArray[i]);
         }
      }
      document.getElementById("outputNum1").value = num1Array.join(",");
      document.getElementById("outputNum2").value = num2Array.join(",");
      document.getElementById("outputNum3").value = num3Array.join(",");

   }

   function downloadableCSV(rows) {
      //var content = "data:text/csv;charset=utf-8,";
      var content = "data:application/csv;charset=utf-8,";
      rows.forEach(function(row, index) {
         content += row + "\n";
      });
      return encodeURI(content);
   }

   function 下载(id){
      var v = document.getElementById(id).value;
      var rows = [];
      rows = v && v.split(",");
      if(rows && rows.length){
         window.open(downloadableCSV(rows));
      }else{
         alert("无内容可下载");
      }
   }
