fun 开通会员提示框() {
   val 弹出框 = AlertDialog.Builder(context);
   val 弹出框布局 = LayoutInflater.from(context).inflate(R.layout.open_vip, null)
   val 关闭 = 弹出框布局.findViewById(R.id.关闭开通会员提示) as ImageButton
   val 确定 = 弹出框布局.findViewById(R.id.开通会员) as Button
   弹出框.setView(弹出框布局)
   val dialog = 弹出框.create()

dialog.setCanceledOnTouchOutside(false);//点击背景不关闭

   关闭.setOnClickListener {
      //todo
      dialog.dismiss()
      if(currentState == JCVideoPlayer.CURRENT_STATE_PAUSE){
         startButton.performClick();
      }
   }
   确定.setOnClickListener {
      val intent = Intent(context, MainActivity::class.java)
      context.startActivity(intent)
   }
   dialog.show()
   dialog.getWindow().setBackgroundDrawable(ColorDrawable(Color.TRANSPARENT))//背景透明
}
