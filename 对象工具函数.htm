对象工具函数

枚举属性：

//将p中的属性添加到o中

function extend(o,p){

for(prop in p){

    o[prop] = p[prop];

}

return o;

}

//将p中的属性添加到o中，o和p中有同名属性，o中的属性不受影响

function merge(o,p){

for(prop in p){

    if(o.hasOwnProperty[prop]) continue;

    o[prop] = p[prop];

}

return o;

}

//如果o中的属性在p中没有同名属性，则从o中删除这个属性

function restrict(o,p){

for(prop in p){

    if(! (prop in p)) delete o[prop];

}

return o;

}

//如果o中的属性在p中存在同名属性，则从o中删除这个属性

function subtract(o,p){

for(prop in p){

    delete o[prop];

}

return o;

}
