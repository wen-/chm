//读取plist 
  NSString *plistPath = [[NSBundle mainBundle] pathForResource:@"万能袋开发-Info" ofType:@"plist"];
  NSMutableDictionary *data = [[NSMutableDictionary alloc] initWithContentsOfFile:plistPath];
  NSLog(@"%@", data);
  NSString *object = [ data objectForKey:@"CodePushDeploymentKey" ];//项


//另一种获取plist文件信息的方法  
  NSDictionary *infoDictionary = [[NSBundle mainBundle] infoDictionary];
 NSLog(@"%@", infoDictionary);

//RN调用方法

RCT_EXPORT_METHOD(config:(RCTResponseSenderBlock)callback) {
  //读取plist信息
  NSDictionary *infoDictionary = [[NSBundle mainBundle] infoDictionary];
  callback(@[infoDictionary]);
}
