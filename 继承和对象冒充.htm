继承和对象冒充

继承

function a(){};
a.prototype={
    b:{
        abc:function(){
            console.log("bbb");
            a.prototype.c();
        }
    },
    c:function(){
        console.log("ccc");
    }
};
var s = new a();
s.b.abc();

对象冒充

function a(){};
a.prototype={
    b:{
        abc:function(){
            console.log("bbb");
            if(this instanceof a){
                this.c();
            }
        }
    },
    c:function(){
        console.log("ccc");
    }
};
var s = new a();
s.b.abc.call(s);
