  1. 匹配中文字符  
  2.   
  3. 普遍使用的正则是[\u4e00-\u9fa5]，但这个范围并不完整。例如：  
  4. /[\u4e00-\u9fa5]/.test( '⻏' ) // 测试部首⻏，返回false  
  5. 根据Unicode 5.0版编码，要准确的判断一个中文字符要包括：  
  6. 范围  含义  范围  含义  
  7. 2E80-2EFF   CJK 部首补充    2F00-2FDF   康熙字典部首  
  8. 3000-303F   CJK 符号和标点   31C0-31EF   CJK 笔画  
  9. 3200-32FF   封闭式 CJK 文字和月份   3300-33FF   CJK 兼容  
  10. 3400-4DBF   CJK 统一表意符号扩展 A  4DC0-4DFF   易经六十四卦符号  
  11. 4E00-9FBF   CJK 统一表意符号  F900-FAFF   CJK 兼容象形文字  
  12. FE30-FE4F   CJK 兼容形式    FF00-FFEF   全角ASCII、全角标点  
  13. 因此，正确的匹配中文字符正则表达式为：  
  14. var rcjk = /[\u2E80-\u2EFF\u2F00-\u2FDF\u3000-\u303F\u31C0-\u31EF\u3200-\u32FF\u3300-\u33FF\u3400-\u4DBF\u4DC0-\u4DFF\u4E00-\u9FBF\uF900-\uFAFF\uFE30-\uFE4F\uFF00-\uFFEF]+/g;  
  15. 如果不希望匹配标点、符号，在正则中去掉对应的范围即可：  
  16. 3000-303F   CJK 符号和标点   FF00-FFEF   全角ASCII、全角标点

  1. 匹配HTML标记的正则表达式  
  2.   
  3. 先说说网上流传的版本：  
  4. <(\S*?)[^>]*>.*?</\1>|<.*? />  
  5. *?  *表示0个或多个，?表示0个或1个，两个叠加起来标识0个多个，与*的功能重叠  
  6. (\S*?)  标签的长度必须大于0，因此不能用*?   
  7. |<.*?\/>  没有分组，无法获取以<div/>这种自关闭格式书写的标签  
  8. </\1>  
  9. <.*? />   有的标签是不关闭的，比如<br><hr>，因此不能强制关闭  
  10. 修正如下：  
  11. var rtag = /^<([a-z]+)\s*\/?>.*(?:<\/\1>)?$/i  
  12. rtag.exec( '<-div></-div>') // null   
  13. rtag.exec( '<div>abc') // ["<div>abc", "div"]  
  14. 这个表达式也不完善，比如第二条测试语句，这么写是为了能提取中包含了文本内容的标签，如果要严格匹配，可再次修改为：  
  15. var rtag = /^<([a-z]+)\s*\/?> (?:<\/\1>)?$/i // 去掉了中间的.*  
  16. 这个正则的应用范围仅限于简单的标签匹配、提取，不能匹配嵌套标记。

  1. 匹配首尾空白字符的正则表达式  
  2.   
  3. 先说说网上流传的版本：  
  4. ^\s*|\s*$  
  5. 可以删除行首行尾的空白字符，例如：  
  6. '   \t    \n\r   abc   \t    \n\r   '.replace( /^\s*|\s*$/g, '' ) // abc  
  7. 但是用\s*不能判断出字符串是否在开头或结尾处有\s，例如：  
  8. /^\s*|\s*$/.test( 'abc' ) // true  
  9. 修正如下：  
  10. ^\s+|\s+$  
  11. '   \t    \n\r   abc   \t    \n\r   '.replace( /^\s+|\s+$/g, '' ) // abc  
  12. /^\s+|\s+$/.test( 'abc' ) // false

  1. 匹配Email地址的正则表达式  
  2.   
  3. 先介绍下Email的规则：local-part@domain  
  4.    local-part最长64，domain最长253，总长最长256  
  5.    local-part可以使用任意ASCII字符：  
  6.    大小写英文字母 a-z,A-Z  
  7.    数字 0-9  
  8.    字符 !#$%&'*+-/=?^_`{|}~  
  9.    字符 .不能是第一个和最后一个，不能连续出现两次  
  10.    但是有些邮件服务器会拒绝包含有特殊字符的邮件地址  
  11.    domain（域名）仅限于26个英文字母、10个数字、连词号-  
  12.    连词号-不能是第一个字符  
  13.    顶级域名（com、cn等）长度为2到6个  
  14. 先说说网上流传的版本：  
  15. \w+([-+.]\w+)*@\w+([-.]\w+)*\.\w+([-.]\w+)*  
  16. ()  莫名奇妙的分组，如果只分组不记录，可以使用(?:)  
  17. @\w domain不能包含下划线_  
  18. \w+([-.]\w+)*   顶级域名不符合规则  
  19. 修正如下：  
  20. var remail = /^([\w-_]+(?:\.[\w-_]+)*)@((?:[a-z0-9]+(?:-[a-zA-Z0-9]+)*)+\.[a-z]{2,6})$/i  
  21. remail.exec( 'nuysoft@gmail.com' ) // "nuysoft@gmail.com", "nuysoft", "gmail.com"]  
  22. remail.exec( 'nuysoft@gmail.comcomcom' ) // null  
  23. remail.exec( 'nuysoft@_gmail.com ) // null  
  24. 修正后的正则有如下局限性：  
  25.    不支持中文邮箱、中文域名，之所以不在其中支持是因为我个人的爱好倾向，反感这类华而不实的玩意  
  26.    不支持特殊符号，避免非邮件服务器拒绝，如果需要，可以添加。  
  27. 参考文章：  
  28. http://en.wikipedia.org/wiki/Email_address

  12. function parseID(ID) {  
  13.     if ( ID.length == 15 ) {  
  14.         // 升级为18位  
  15.         ID = ID.substr( 0, 6 ) + "19" + ID.substr( 6 );  
  16.         // 前17位对应的系数  
  17.         var rank = [  
  18.                 "7", "9", "10", "5", "8", "4", "2", "1", "6", "3", "7", "9", "10", "5", "8", "4", "2"  
  19.         ];  
  20.         // 前17为加权除以17后的余数对应的最后一位身份证号码  
  21.         var last = [  
  22.                 "1", "0", "X", "9", "8", "7", "6", "5", "4", "3", "2"  
  23.         ];  
  24.         // 加权和  
  25.         for ( var i = 0, sum = 0, len = ID.length; i < len; i++)  
  26.             sum += ID[ i ] * rank[ i ];  
  27.         // 加上最后一位  
  28.         ID += last[ sum % 11 ];  
  29.     }  
  30.     if ( ID.length != 18 ) return null;  
  31.   
  32.     var match = rid.exec( ID );  
  33.     return match ? {  
  34.         ID : ID,  
  35.         area : match[ 1 ],

  37.         m : match[ 3 ],  
  38.         d : match[ 4 ],  
  39.         sex : match[ 5 ] % 2
