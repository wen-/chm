上面的就是不间断滚动了，翻译下这段代码
<script>
var speed=30; //设变量 滚动速度变量speed =30 
moved2.innercode=moved1.innercode //复制moved1的code代码给moved2

function Marquee()
{
if(moved2.offsetTop-moved.scrollTop<=0) { //如果moved2.offset-moved.scrolltop<=0(也就是moved1的内容滚动结束)则moved2开始滚动
moved.scrollTop-=moved1.offsetHeight; //moved.scrolltop此时的值为moved2滚动的长度
}
else {
moved.scrollTop++; //否则moved1继续滚动
}
}
var MyMar=setInterval(Marquee,speed) //每隔30毫秒 执行一次
moved.onmouseover = function() {clearInterval(MyMar);} //鼠标移过 停止执行
moved.onmouseout = function() {MyMar=setInterval(Marquee,speed);} //滑出 继续执行
</script>
