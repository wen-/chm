属性getter和属性setter

var p={

x:1.0,

y:1.0,

//r是可读可写的存取器属性

get r(){return Math.sqrt(this.x*this.x + this.y*this.y)},

set r(newvalue){var oldvalue = Math.sqrt(this.x*this.x + this.y*this.y);

var ratio = newvalue/oldvalue;

this.x*=ratio;

this.y*=ratio;

},

//theta是只读存取器属性

get theta(){return Math.atan2(this.y,this.x);}

};
